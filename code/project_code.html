import pandas as pd
import numpy as np
from scipy import stats

df = pd.read_csv("projectdata.csv")

# Create education dummies
edu_dummies = pd.get_dummies(df['educ'], prefix='educ')
df = pd.concat([df, edu_dummies], axis=1)

# VARIABLES REQUIRED BY ASSIGNMENT
vars_used = ["age", "educ", "y", "owage2",
             "educ_6","educ_9","educ_12","educ_16"]

# --- Summary stats ---
summary_all = df[vars_used].mean()
summary_female = df[df.female==1][vars_used].mean()
summary_male = df[df.female==0][vars_used].mean()

# --- t-stats for continuous variables only ---
t_stats = {}
for v in ["age","educ","y","owage2"]:
    t_stats[v] = stats.ttest_ind(
        df[df.female==1][v],
        df[df.female==0][v],
        equal_var=False
    )[0]   # ONLY t-stat

# Construct clean table
table1 = pd.DataFrame({
    "All": summary_all,
    "Female": summary_female,
    "Male": summary_male,
})

# Add t-stat column (match only continuous rows)
table1["t_stat"] = table1.index.map(lambda x: t_stats.get(x, np.nan))
